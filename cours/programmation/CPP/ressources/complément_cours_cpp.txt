/* notion de récursivité : "qui se répète tout seul indéfiniment"
/-> factorielle (n!): 
= 1 si n =< 1
= n * (n-1)! pour le reste
= 1 * 2 * 3 * 4 * ... * n -1 * n

int factorielle(int n) {
    if(n =< 1) return 1;
    else return n * fact(n - 1);
}
